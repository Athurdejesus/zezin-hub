-- ==================== FUNÇÃO PARA TORNAR TUDO ROXO ====================
local function tornarTudoRoxo()
    local roxoEscuro = Color3.new(0.3, 0, 0.5)
    local roxoClaro = Color3.new(0.8, 0.4, 1)
    local roxoVibrante = Color3.new(0.6, 0, 1)
    
    task.wait(0.5)
    
    for _, gui in ipairs(player.PlayerGui:GetChildren()) do
        if gui:IsA("ScreenGui") and gui.Name:find("Kavo") then
            local function mudarCores(obj)
                if obj:IsA("Frame") then
                    if obj.BackgroundColor3 == Color3.new(0, 0, 0) or obj.BackgroundColor3 == Color3.new(0.1, 0.1, 0.1) then
                        obj.BackgroundColor3 = roxoEscuro
                    end
                end
                
                if obj:IsA("TextLabel") or obj:IsA("TextButton") then
                    if obj.TextColor3 == Color3.new(1, 1, 1) then
                        obj.TextColor3 = roxoClaro
                    end
                end
                
                if obj:IsA("UIStroke") then
                    obj.Color = roxoVibrante
                end
                
                if obj:IsA("ScrollingFrame") then
                    obj.ScrollBarImageColor3 = roxoVibrante
                end
                
                for _, child in ipairs(obj:GetChildren()) do
                    mudarCores(child)
                end
            end
            mudarCores(gui)
            break
        end
    end
end

spawn(tornarTudoRoxo)

-- ==================== ESP ====================
local espObjects = {}

local function criarESP(jogador)
    if jogador == player then return end
    
    local function onCharacter(character)
        local head = character:WaitForChild("Head", 5)
        local humanoid = character:WaitForChild("Humanoid", 5)
        
        if not head or not humanoid then return end
        
        local billboard = Instance.new("BillboardGui")
        billboard.Name = "ESP_" .. jogador.Name
        billboard.Adornee = head
        billboard.Size = UDim2.new(0, 150, 0, 50)
        billboard.StudsOffset = Vector3.new(0, 3, 0)
        billboard.AlwaysOnTop = true
        billboard.Parent = character
        
        local nomeLabel = Instance.new("TextLabel")
        nomeLabel.Parent = billboard
        nomeLabel.BackgroundTransparency = 1
        nomeLabel.Size = UDim2.new(1, 0, 0.5, 0)
        nomeLabel.Text = jogador.Name
        nomeLabel.TextColor3 = espColor
        nomeLabel.TextScaled = true
        nomeLabel.Font = Enum.Font.GothamBold
        
        local vidaLabel = Instance.new("TextLabel")
        vidaLabel.Parent = billboard
        vidaLabel.BackgroundTransparency = 1
        vidaLabel.Size = UDim2.new(1, 0, 0.5, 0)
        vidaLabel.Position = UDim2.new(0, 0, 0.5, 0)
        vidaLabel.Text = "❤️ " .. math.floor(humanoid.Health)
        vidaLabel.TextColor3 = Color3.new(0, 1, 0)
        vidaLabel.TextScaled = true
        vidaLabel.Font = Enum.Font.GothamBold
        
        local highlight = Instance.new("Highlight")
        highlight.Name = "HL_" .. jogador.Name
        highlight.Adornee = character
        highlight.FillColor = espColor
        highlight.FillTransparency = 0.5
        highlight.OutlineColor = Color3.new(1, 1, 1)
        highlight.Parent = character
        
        table.insert(espObjects, {
            billboard = billboard,
            highlight = highlight,
            jogador = jogador
        })
    end
    
    if jogador.Character then
        onCharacter(jogador.Character)
    end
    
    jogador.CharacterAdded:Connect(onCharacter)
end

-- ==================== ESP LINHA ====================
local linhasESP = {}

local function criarLinhaESP(jogador)
    if jogador == player then return end
    
    local linha = Drawing.new("Line")
    linha.Visible = false
    linha.Color = espColor
    linha.Thickness = 1
    linha.Transparency = 0.5
    
    table.insert(linhasESP, {
        linha = linha,
        jogador = jogador
    })
end

local function atualizarLinhasESP()
    if not espLinhaEnabled then
        for _, l in ipairs(linhasESP) do
            if l.linha then
                l.linha.Visible = false
            end
        end
        return
    end
    
    local telaCentro = Vector2.new(camera.ViewportSize.X / 2, camera.ViewportSize.Y)
    
    for _, l in ipairs(linhasESP) do
        local jogador = l.jogador
        local linha = l.linha
        
        if jogador and jogador.Character and jogador.Character:FindFirstChild("HumanoidRootPart") and jogador.Character:FindFirstChildOfClass("Humanoid") and jogador.Character.Humanoid.Health > 0 then
            local root = jogador.Character.HumanoidRootPart
            local posJogador, naTela = camera:WorldToViewportPoint(root.Position)
            
            if naTela then
                linha.Visible = true
                linha.From = Vector2.new(telaCentro.X, telaCentro.Y)
                linha.To = Vector2.new(posJogador.X, posJogador.Y)
                linha.Color = espColor
            else
                linha.Visible = false
            end
        else
            linha.Visible = false
        end
    end
end

runService.Heartbeat:Connect(atualizarLinhasESP)

-- ==================== FUNÇÕES ====================
local function getPlayerList()
    local list = {}
    for _, v in ipairs(players:GetPlayers()) do
        if v ~= player then
            table.insert(list, v.Name)
        end
    end
    return list
end

local function getPlayer(name)
    for _, v in ipairs(players:GetPlayers()) do
        if v.Name == name then
            return v
        end
    end
    return nil
end

-- ANTI RAGDOLL CORRIGIDO
local function checkAntiRagdoll()
    if not antiRagdollEnabled or not player.Character then return end
    
    local humanoid = player.Character:FindFirstChildOfClass("Humanoid")
    if not humanoid then return end
    
    local ragdollStates = {
        [Enum.HumanoidStateType.FallingDown] = true,
        [Enum.HumanoidStateType.Ragdoll] = true,
        [Enum.HumanoidStateType.Physics] = true,
        [Enum.HumanoidStateType.Dead] = true,
        [Enum.HumanoidStateType.Freefall] = true
    }
    
    if ragdollStates[humanoid:GetState()] then
        humanoid:ChangeState(Enum.HumanoidStateType.Running)
        
        if player.Character:FindFirstChild("RagdollConstraint") then
            player.Character.RagdollConstraint:Destroy()
        end
        
        local rootPart = player.Character:FindFirstChild("HumanoidRootPart")
        if rootPart then
            rootPart.Velocity = Vector3.new(0, 10, 0)
        end
    end
end

runService.Heartbeat:Connect(function()
    if antiRagdollEnabled then
        checkAntiRagdoll()
    end
end)

-- ==================== FUNÇÃO PARA TORNAR TUDO ROXO ====================
local function tornarTudoRoxo()
    local roxoEscuro = Color3.new(0.3, 0, 0.5)
    local roxoClaro = Color3.new(0.8, 0.4, 1)
    local roxoVibrante = Color3.new(0.6, 0, 1)
    
    task.wait(0.5)
    
    for _, gui in ipairs(player.PlayerGui:GetChildren()) do
        if gui:IsA("ScreenGui") and gui.Name:find("Kavo") then
            local function mudarCores(obj)
                if obj:IsA("Frame") then
                    if obj.BackgroundColor3 == Color3.new(0, 0, 0) or obj.BackgroundColor3 == Color3.new(0.1, 0.1, 0.1) then
                        obj.BackgroundColor3 = roxoEscuro
                    end
                end
                
                if obj:IsA("TextLabel") or obj:IsA("TextButton") then
                    if obj.TextColor3 == Color3.new(1, 1, 1) then
                        obj.TextColor3 = roxoClaro
                    end
                end
                
                if obj:IsA("UIStroke") then
                    obj.Color = roxoVibrante
                end
                
                if obj:IsA("ScrollingFrame") then
                    obj.ScrollBarImageColor3 = roxoVibrante
                end
                
                for _, child in ipairs(obj:GetChildren()) do
                    mudarCores(child)
                end
            end
            mudarCores(gui)
            break
        end
    end
end

spawn(tornarTudoRoxo)

-- ==================== ESP ====================
local espObjects = {}

local function criarESP(jogador)
    if jogador == player then return end
    
    local function onCharacter(character)
        local head = character:WaitForChild("Head", 5)
        local humanoid = character:WaitForChild("Humanoid", 5)
        
        if not head or not humanoid then return end
        
        local billboard = Instance.new("BillboardGui")
        billboard.Name = "ESP_" .. jogador.Name
        billboard.Adornee = head
        billboard.Size = UDim2.new(0, 150, 0, 50)
        billboard.StudsOffset = Vector3.new(0, 3, 0)
        billboard.AlwaysOnTop = true
        billboard.Parent = character
        
        local nomeLabel = Instance.new("TextLabel")
        nomeLabel.Parent = billboard
        nomeLabel.BackgroundTransparency = 1
        nomeLabel.Size = UDim2.new(1, 0, 0.5, 0)
        nomeLabel.Text = jogador.Name
        nomeLabel.TextColor3 = espColor
        nomeLabel.TextScaled = true
        nomeLabel.Font = Enum.Font.GothamBold
        
        local vidaLabel = Instance.new("TextLabel")
        vidaLabel.Parent = billboard
        vidaLabel.BackgroundTransparency = 1
        vidaLabel.Size = UDim2.new(1, 0, 0.5, 0)
        vidaLabel.Position = UDim2.new(0, 0, 0.5, 0)
        vidaLabel.Text = "❤️ " .. math.floor(humanoid.Health)
        vidaLabel.TextColor3 = Color3.new(0, 1, 0)
        vidaLabel.TextScaled = true
        vidaLabel.Font = Enum.Font.GothamBold
        
        local highlight = Instance.new("Highlight")
        highlight.Name = "HL_" .. jogador.Name
        highlight.Adornee = character
        highlight.FillColor = espColor
        highlight.FillTransparency = 0.5
        highlight.OutlineColor = Color3.new(1, 1, 1)
        highlight.Parent = character
        
        table.insert(espObjects, {
            billboard = billboard,
            highlight = highlight,
            jogador = jogador
        })
    end
    
    if jogador.Character then
        onCharacter(jogador.Character)
    end
    
    jogador.CharacterAdded:Connect(onCharacter)
end

-- ==================== ESP LINHA ====================
local linhasESP = {}

local function criarLinhaESP(jogador)
    if jogador == player then return end
    
    local linha = Drawing.new("Line")
    linha.Visible = false
    linha.Color = espColor
    linha.Thickness = 1
    linha.Transparency = 0.5
    
    table.insert(linhasESP, {
        linha = linha,
        jogador = jogador
    })
end

local function atualizarLinhasESP()
    if not espLinhaEnabled then
        for _, l in ipairs(linhasESP) do
            if l.linha then
                l.linha.Visible = false
            end
        end
        return
    end
    
    local telaCentro = Vector2.new(camera.ViewportSize.X / 2, camera.ViewportSize.Y)
    
    for _, l in ipairs(linhasESP) do
        local jogador = l.jogador
        local linha = l.linha
        
        if jogador and jogador.Character and jogador.Character:FindFirstChild("HumanoidRootPart") and jogador.Character:FindFirstChildOfClass("Humanoid") and jogador.Character.Humanoid.Health > 0 then
            local root = jogador.Character.HumanoidRootPart
            local posJogador, naTela = camera:WorldToViewportPoint(root.Position)
            
            if naTela then
                linha.Visible = true
                linha.From = Vector2.new(telaCentro.X, telaCentro.Y)
                linha.To = Vector2.new(posJogador.X, posJogador.Y)
                linha.Color = espColor
            else
                linha.Visible = false
            end
        else
            linha.Visible = false
        end
    end
end

runService.Heartbeat:Connect(atualizarLinhasESP)

-- ANTI RAGDOLL CORRIGIDO
local function checkAntiRagdoll()
    if not antiRagdollEnabled or not player.Character then return end
    
    local humanoid = player.Character:FindFirstChildOfClass("Humanoid")
    if not humanoid then return end
    
    local ragdollStates = {
        [Enum.HumanoidStateType.FallingDown] = true,
        [Enum.HumanoidStateType.Ragdoll] = true,
        [Enum.HumanoidStateType.Physics] = true,
        [Enum.HumanoidStateType.Dead] = true,
        [Enum.HumanoidStateType.Freefall] = true
    }
    
    if ragdollStates[humanoid:GetState()] then
        humanoid:ChangeState(Enum.HumanoidStateType.Running)
        
        if player.Character:FindFirstChild("RagdollConstraint") then
            player.Character.RagdollConstraint:Destroy()
        end
        
        local rootPart = player.Character:FindFirstChild("HumanoidRootPart")
        if rootPart then
            rootPart.Velocity = Vector3.new(0, 10, 0)
        end
    end
end

runService.Heartbeat:Connect(function()
    if antiRagdollEnabled then
        checkAntiRagdoll()
    end
end)
